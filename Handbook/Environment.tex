% Embed colour profile information in PDF to make consistent across viewers. 
%  This is a temporary workaround to solve this problem:
%  <http://archive.contextgarden.net/message/20120415.060028.20f62826.en.html>
%  Note: Resolved upstream in latest beta. Use \setupcolors[pagecolormodel=rgb]

    \startluacode
    backends                = backends                or { }
    backends.codeinjections = backends.codeinjections or { }

    function backends.codeinjections.rgbtransparencygroup()
        local d = lpdf.dictionary {
            S  = lpdf.constant("Transparency"),
            CS = lpdf.constant("DeviceRGB"),
            I  = true }
        lpdf.registerpagefinalizer(function()
    lpdf.addtopageattributes("Group",d) end)
    end

    backends.codeinjections.rgbtransparencygroup()
    \stopluacode
