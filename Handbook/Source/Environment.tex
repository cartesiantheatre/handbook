% Insert a full page landscape diagram, parameters are node name, caption, and path...
\def\FullPageLandscapeDiagram#1#2#3%
{
    \page 
    \setuppapersize[diagram]
    \setuplayout[fullpage]
    \placefigure
        [force][figure:#1]
        {#2}
        {\externalfigure[#3][factor=max]}
    \page
    \setuppapersize[main]
    \setuplayout[reset]
}
